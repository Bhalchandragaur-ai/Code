This program is to Check That It is a Character, Digit or Special Character.

C Code:

#include<stdio.h>

int main()
{
    char ch;
    printf("Enter a Character:");
    scanf("%c",&ch);
    if ((ch>='a' && ch<='z')||(ch>='A' && ch<='Z'))
        printf("It is an Alphabet");
    else if (ch>='0' && ch<='9')
        printf("It is a digit");
    else
        printf("It is a Special Character");
    return 0;
}
